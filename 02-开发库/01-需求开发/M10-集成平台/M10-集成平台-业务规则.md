#### BR-INT-VAL-002: 身份证号格式校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-INT-VAL-002 |
| 规则名称 | 身份证号格式校验 |
| 规则类型 | 校验规则 (VAL) |
| 适用对象 | EMPI主索引(empi) |
| 规则描述 | 身份证号必须符合18位国家标准格式，包含校验位验证 |
| 触发时机 | EMPI创建/更新时 |
| 校验逻辑 | ```
// 伪代码
IF id_card_no IS NOT NULL AND id_card_no != '' THEN
    // 长度校验
    IF LENGTH(id_card_no) != 18 THEN
        RETURN error("身份证号长度不正确，应为18位")
    END IF
    
    // 格式校验
    IF NOT match_regex(id_card_no, '^[1-9][0-9]{16}[0-9Xx]$') THEN
        RETURN error("身份证号格式不正确")
    END IF
    
    // 校验位验证
    weights = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2]
    check_codes = ['1', '0', 'X', '9', '8', '7', '6', '5', '4', '3', '2']
    
    sum = 0
    FOR i FROM 0 TO 16 DO
        sum = sum + INTEGER(id_card_no[i]) * weights[i]
    END FOR
    
    check_code = check_codes[sum % 11]
    IF UPPER(id_card_no[17]) != check_code THEN
        RETURN error("身份证号校验位不正确")
    END IF
END IF

RETURN success()
``` |
| 失败处理 | 返回具体校验失败原因，阻止EMPI保存 |
| 优先级 | P0 |

FUNCTION CALCULATE_NAME_SIMILARITY(name1, name2)
    // 完全相同
    IF name1 = name2 THEN
        RETURN 1.0
    END IF
    
    // 标准化处理
    name1 = NORMALIZE(name1)  // 去空格、转大写
    name2 = NORMALIZE(name2)
    
    // 编辑距离
    edit_distance = LEVENSHTEIN_DISTANCE(name1, name2)
    max_length = MAX(LENGTH(name1), LENGTH(name2))
    
    // 相似度
    similarity = 1.0 - (edit_distance / max_length)
    
    // 中文特殊处理
    IF IS_CHINESE(name1) AND IS_CHINESE(name2) THEN
        // 考虑同音字、形近字
        pronunciation_similarity = COMPARE_PRONUNCIATION(name1, name2)
        similarity = MAX(similarity, pronunciation_similarity)
    END IF
    
    RETURN similarity
END FUNCTION

// 伪代码
FUNCTION SEND_ALERT(alert)
    // 计算告警指纹
    fingerprint = HASH(alert.interface_id + alert.alert_type + alert.alert_level)
    
    // 检查最近是否有相似告警
    recent_similar = SELECT count(*) FROM alert_history 
                     WHERE fingerprint = :fingerprint 
                     AND create_time > NOW() - INTERVAL 5 MINUTE
    
    IF recent_similar > 0 THEN
        // 已有相似告警，合并发送
        UPDATE alert_history SET
            count = count + 1,
            last_occurrence = NOW()
        WHERE fingerprint = :fingerprint
        AND create_time > NOW() - INTERVAL 5 MINUTE
        
        // 每累计10条发送一次汇总
        IF recent_similar % 10 = 0 THEN
            SEND_SUMMARY_ALERT(fingerprint)
        END IF
        
        RETURN "MERGED"
    ELSE
        // 新告警，正常发送
        INSERT INTO alert_history (...)
        VALUES (...)
        
        SEND_ALERT_NOTIFICATION(alert)
        
        RETURN "SENT"
    END IF
END FUNCTION

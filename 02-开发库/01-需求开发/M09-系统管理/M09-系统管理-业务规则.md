#### BR-M09-VAL-003: 密码复杂度校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-M09-VAL-003 |
| 规则名称 | 密码复杂度校验 |
| 规则类型 | 校验规则(VAL) |
| 适用对象 | 用户密码(password) |
| 规则描述 | 密码必须满足复杂度要求：长度≥8位，包含大小写字母、数字、特殊字符中的至少3类 |
| 触发条件 | 用户创建、密码修改、密码重置时 |
| 执行逻辑 | ```
// 伪代码
password = :new_password

// 长度校验
IF LENGTH(password) < 8 THEN
    RETURN error("密码长度不能少于8位")
END IF

// 复杂度计算
complexity_count = 0
IF password CONTAINS '[a-z]' THEN complexity_count++ END IF  // 小写字母
IF password CONTAINS '[A-Z]' THEN complexity_count++ END IF  // 大写字母
IF password CONTAINS '[0-9]' THEN complexity_count++ END IF  // 数字
IF password CONTAINS '[!@#$%^&*()_+\\-=\\[\\]{};\':"\\\\|,.<>\\/?]' THEN complexity_count++ END IF  // 特殊字符

IF complexity_count < 3 THEN
    RETURN error("密码复杂度不足，必须包含大写字母、小写字母、数字、特殊字符中的至少3类")
END IF

// 禁止与用户名相同或包含用户名
IF password = :username OR password CONTAINS :username THEN
    RETURN error("密码不能与用户名相同或包含用户名")
END IF

// 禁止常见弱密码
weak_passwords = ['123456', 'password', 'qwerty', 'abc123', ...]
IF password IN weak_passwords THEN
    RETURN error("密码过于简单，请设置更复杂的密码")
END IF

RETURN success()
``` |
| 异常处理 | 返回具体复杂度不足原因，提示密码要求 |
| 优先级 | P0 |

#### BR-M09-VAL-006: 身份证号格式校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-M09-VAL-006 |
| 规则名称 | 身份证号格式校验 |
| 规则类型 | 校验规则(VAL) |
| 适用对象 | 用户身份证号(id_card) |
| 规则描述 | 身份证号必须符合18位身份证格式，包含校验位验证 |
| 触发条件 | 填写或修改身份证号时 |
| 执行逻辑 | ```
// 伪代码
IF id_card IS NOT NULL AND id_card != '' THEN
    // 基本格式校验
    IF NOT match_regex(id_card, '^[1-9][0-9]{5}(19|20)[0-9]{2}(0[1-9]|1[0-2])(0[1-9]|[12][0-9]|3[01])[0-9]{3}[0-9Xx]$') THEN
        RETURN error("身份证号格式不正确")
    END IF
    
    // 校验位验证
    weights = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2]
    check_codes = ['1', '0', 'X', '9', '8', '7', '6', '5', '4', '3', '2']
    sum = 0
    FOR i = 0 TO 16 DO
        sum = sum + INT(id_card[i]) * weights[i]
    END FOR
    check_code = check_codes[sum % 11]
    
    IF UPPER(id_card[17]) != check_code THEN
        RETURN error("身份证号校验位不正确，请核实身份证号")
    END IF
END IF

RETURN success()
``` |
| 异常处理 | 返回错误提示"身份证号格式不正确" |
| 优先级 | P1 |

    RETURN error("父菜单不存在")

IF new_level > 3 THEN
    RETURN error("菜单层级不能超过3级")

IF :type = 2 AND parent_menu.type = 3 THEN
    RETURN error("菜单不能创建在按钮下")

    RETURN error("按钮只能创建在菜单下")
END IF

#### BR-M09-FLOW-002: 登录失败锁定机制

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-M09-FLOW-002 |
| 规则名称 | 登录失败锁定机制 |
| 规则类型 | 流转规则(FLOW) |
| 适用对象 | 用户登录(login) |
| 规则描述 | 连续登录失败5次锁定账号30分钟，锁定期间禁止登录 |
| 触发条件 | 用户登录失败时 |
| 执行逻辑 | ```

// 检查是否已锁定
IF user.status = 3 THEN
    // 检查锁定时间是否超过30分钟
    lock_duration = MINUTE(NOW() - user.lock_time)
    IF lock_duration >= 30 THEN
        // 自动解锁
        UPDATE sys_user SET status = 1, fail_count = 0, lock_time = NULL
        WHERE user_id = user.user_id
        user.status = 1
    ELSE
        remaining_minutes = 30 - lock_duration
        RETURN error("账号已锁定，请" + remaining_minutes + "分钟后重试")
    END IF
END IF

// 密码校验
IF NOT verify_password(:password, user.password) THEN
    // 登录失败，增加失败计数
    new_fail_count = user.fail_count + 1
    update_data = { 'fail_count': new_fail_count }
    
    IF new_fail_count >= 5 THEN
        // 达到5次，锁定账号
        update_data.status = 3
        update_data.lock_time = NOW()
        RETURN error("密码错误次数过多，账号已锁定30分钟")
    ELSE
        remaining_attempts = 5 - new_fail_count
        RETURN error("密码错误，还剩" + remaining_attempts + "次尝试机会")
    END IF
    
    UPDATE sys_user SET ... WHERE user_id = user.user_id
END IF

// 登录成功，清空失败计数
UPDATE sys_user SET fail_count = 0, lock_time = NULL, login_ip = :ip, login_date = NOW()
WHERE user_id = user.user_id

RETURN success()

    WHEN 5 THEN  // 仅本人数据

| 适用对象 | JWT Token(jwt_token) |
| 规则描述 | 登录成功后生成JWT Token，包含用户信息和权限，有效期2小时 |
| 触发条件 | 用户登录成功时、接口调用验证时 |
| 执行逻辑 | ```
// 伪代码
// Token配置
TOKEN_SECRET = GET_config('sys.token.secret')  // 密钥
TOKEN_EXPIRE = GET_config('sys.token.expire', 7200)  // 有效期（秒）

// Token生成
FUNCTION generate_token(user_id):
    // 1. 获取用户信息和权限
    user = SELECT user_id, username, real_name, dept_id, tenant_id 
           FROM sys_user WHERE user_id = user_id
    permissions = GET_user_permissions(user_id)
    
    // 2. 构建Token载荷
    payload = {
        'user_id': user.user_id,
        'username': user.username,
        'real_name': user.real_name,
        'dept_id': user.dept_id,
        'tenant_id': user.tenant_id,
        'permissions': permissions,
        'iat': NOW_TIMESTAMP,  // 签发时间
        'exp': NOW_TIMESTAMP + TOKEN_EXPIRE  // 过期时间
    }
    
    // 3. 生成JWT Token
    token = JWT.encode(payload, TOKEN_SECRET, algorithm='HS512')
    
    // 4. 存储Token到Redis（用于主动失效）
    REDIS.setex('token:' + user_id, TOKEN_EXPIRE, token)
    
    RETURN token

// Token验证
FUNCTION verify_token(token):
    TRY
        // 1. 解码Token
        payload = JWT.decode(token, TOKEN_SECRET, algorithms=['HS512'])
        
        // 2. 检查是否过期
        IF payload.exp < NOW_TIMESTAMP THEN
            RETURN error("Token已过期")
        END IF
        
        // 3. 检查Redis中是否有效（支持主动失效）
        stored_token = REDIS.get('token:' + payload.user_id)
        IF stored_token IS NULL OR stored_token != token THEN
            RETURN error("Token已失效")
        END IF
        
        RETURN success(payload)
    CATCH JWT.DecodeError
        RETURN error("Token无效")
    END TRY

            FROM sys_role r

        RETURN true
    END IF

    wildcards = [
        '*:*:*',
        permission_parts[0] + ':*:*',
        permission_parts[0] + ':' + permission_parts[1] + ':*'
    ]
    
    RETURN ANY wildcard IN wildcards WHERE wildcard IN permissions

FUNCTION calculate_data_scope(user_id):

                  FROM sys_role r

            FOR each role IN role_scopes WHERE role.data_scope = 2:
                custom_depts = custom_depts + PARSE dept_ids
            END FOR

            // 获取科室树
            dept_tree = GET_dept_and_children(user.dept_id)
            sql = "dept_id IN (" + dept_tree + ")"
            
        WHEN 5 THEN  // 仅本人
            sql = "create_by = " + user_id
    END CASE
    
    RETURN { 'scope': min_scope, 'sql': sql }

FUNCTION ca_login(ca_certificate):
    // 1. 解析CA证书

    CATCH CA.Error
        RETURN error("CA证书解析失败")
    END TRY

    validation_result = CA.validate_certificate(cert_info)

    // 6. 生成Token

    RETURN success(token)

        'mobile': 'phone',
        'email': 'email',
        'dept_code': 'dept_code',
        'status': 'status'

FUNCTION sync_users_from_external():
    // 1. 从外部系统获取用户列表
    TRY
        external_users = EXTERNAL_SYSTEM.get_users()
    CATCH Exception AS e
        LOG.error("外部系统连接失败：" + e.message)
        RETURN error("同步失败")
    END TRY
    
    // 2. 获取内部用户工号列表
    internal_users = SELECT user_no, user_id FROM sys_user WHERE is_deleted = 0
    
    // 3. 遍历处理
    stats = { 'created': 0, 'updated': 0, 'disabled': 0, 'skipped': 0 }
    
    FOR each ext_user IN external_users:
        mapped_user = MAP_FIELDS(ext_user, sync_config.mapping)
        
        existing = FIND internal_users WHERE user_no = mapped_user.user_no
        
        IF existing IS NULL THEN
            // 新用户，创建
            IF sync_config.auto_create THEN
                new_user = CREATE_user(mapped_user)
                new_user.password = BCrypt.hash(GENERATE_random_password())
                new_user.source = sync_config.source
                SAVE new_user
                stats.created++
            END IF
        ELSE
            // 已存在，更新
            CASE sync_config.conflict_resolution
                WHEN 'UPDATE' THEN
                    UPDATE existing SET ... FROM mapped_user
                    stats.updated++
                WHEN 'OVERWRITE' THEN
                    OVERWRITE existing FROM mapped_user
                    stats.updated++
                WHEN 'SKIP' THEN
                    stats.skipped++
            END CASE
        END IF
    END FOR
    
    // 4. 处理离职用户（在外部系统中不存在但内部存在）
    IF sync_config.auto_disable THEN
        external_nos = [ext_user.employee_id FOR ext_user IN external_users]
        FOR each internal IN internal_users:
            IF internal.user_no NOT IN external_nos THEN
                UPDATE sys_user SET status = 2 WHERE user_id = internal.user_id
                stats.disabled++
            END IF
        END FOR
    END IF
    
    // 5. 记录同步日志
    INSERT INTO sys_sync_log (source, sync_time, created, updated, disabled, skipped)
    VALUES (sync_config.source, NOW(), stats.created, stats.updated, stats.disabled, stats.skipped)
    
    RETURN success(stats)

    signature = request.header('X-Signature')
    
    // 2. 参数校验
    IF app_key IS NULL OR timestamp IS NULL OR signature IS NULL THEN
        RETURN error("缺少认证参数", status=401)
    END IF
    
    // 3. 时间戳校验（防重放攻击）
    IF ABS(NOW_TIMESTAMP - timestamp) > 300 THEN  // 5分钟有效期
        RETURN error("请求已过期", status=401)
    END IF
    
    // 4. 查询App信息
    app = SELECT * FROM sys_api_app 
          WHERE app_key = app_key AND status = 1
    
    IF app IS NULL THEN
        RETURN error("无效的AppKey", status=401)
    END IF
    
    // 5. 验证签名
    // 签名算法：signature = HMAC-SHA256(app_secret, method + url + timestamp + nonce + body)
    sign_content = request.method + request.url + timestamp + nonce + request.body
    expected_signature = HMAC_SHA256(app.app_secret, sign_content)
    
    IF signature != expected_signature THEN
        LOG.warn("签名验证失败，AppKey：" + app_key)
        RETURN error("签名验证失败", status=401)
    END IF
    
    // 6. 检查接口权限
    api_permission = SELECT * FROM sys_api_permission
                     WHERE app_id = app.app_id
                       AND api_path = request.path
                       AND status = 1
    
    IF api_permission IS NULL THEN
        RETURN error("无接口访问权限", status=403)
    END IF
    
    // 7. 记录调用日志
    INSERT INTO sys_api_log (app_id, api_path, request_time, ip_address)
    VALUES (app.app_id, request.path, NOW(), request.ip)
    
    // 8. 更新调用统计
    UPDATE sys_api_app 
    SET call_count = call_count + 1, last_call_time = NOW()
    WHERE app_id = app.app_id
    
    RETURN success(app)
``` |
| 异常处理 | 认证失败返回401，权限不足返回403 |

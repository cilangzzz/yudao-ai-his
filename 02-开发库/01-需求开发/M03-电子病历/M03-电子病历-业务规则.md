#### BR-EMR-003: 病历必填项校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-EMR-003 |
| 规则名称 | 病历必填项校验 |
| 规则类型 | VAL (校验规则) |
| 适用对象 | 病历文书(emr_document) |
| 规则描述 | 不同类型病历文书有不同的必填项要求,提交审签前必须校验必填项完整性 |
| 触发时机 | 病历提交审签时 |
| 校验逻辑 | ```
// 伪代码
CASE document.document_type
    WHEN 'EMR-ADMISSION' THEN  // 入院记录
        required_fields = ['主诉', '现病史', '既往史', '体格检查', '初步诊断']
        IF LENGTH(主诉) < 10 THEN
            RETURN error("主诉内容不能少于10字符")
        END IF
        IF LENGTH(现病史) < 100 THEN
            RETURN error("现病史内容不能少于100字符")
        END IF
        IF 诊断 IS NULL OR NOT has_icd10_code(诊断) THEN
            RETURN error("初步诊断必须关联ICD-10编码")
        END IF
        
    WHEN 'EMR-FIRST-PROGRESS' THEN  // 首次病程记录
        required_fields = ['病例特点', '诊断依据', '鉴别诊断', '诊疗计划']
        IF LENGTH(病例特点) < 50 THEN
            RETURN error("病例特点内容不能少于50字符")
        END IF
        
    WHEN 'EMR-DISCHARGE' THEN  // 出院记录
        required_fields = ['入院诊断', '诊疗经过', '出院诊断', '出院医嘱']
END CASE

// 通用必填项检查
FOR each field IN required_fields DO
    IF document[field] IS NULL OR document[field] = '' THEN
        RETURN error(field + "为必填项,请补充完整")
    END IF
END FOR

RETURN success()
``` |
| 失败处理 | 返回具体缺失的必填项,阻止提交审签 |
| 优先级 | P0 |

| 校验逻辑 | ```
// 伪代码
// 获取操作时间
operation_time = GET_operation_time(operation_id)
current_time = NOW()

// 检查知情同意书
consent = SELECT * FROM emr_consent 
          WHERE operation_id = :operation_id 
            AND consent_status = 'SIGNED'

IF consent IS NULL THEN
    // 未签署知情同意书
    time_until_operation = operation_time - current_time
    
    IF time_until_operation <= 0 THEN
        RETURN error("操作前必须签署知情同意书")

    RETURN error("知情同意书未完成患者/监护人签名")
END IF

IF consent.doctor_signature IS NULL THEN
    RETURN error("知情同意书未完成医师签名")
END IF

RETURN success()

#### BR-EMR-005: 病历内容长度校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-EMR-005 |
| 规则名称 | 病历内容长度校验 |
| 规则类型 | VAL (校验规则) |
| 适用对象 | 病历文书(emr_document) |
| 规则描述 | 各类病历文书内容有最小字符要求,保证病历质量 |

| 校验逻辑 | ```
// 伪代码
min_length_rules = {
    'EMR-ADMISSION': {
        '主诉': 10,
        '现病史': 100,
        '既往史': 20
    },
    'EMR-FIRST-PROGRESS': {
        '病例特点': 50,
        '诊断依据': 30,
        '鉴别诊断': 30,
        '诊疗计划': 20
    },
    'EMR-DAILY-PROGRESS': {
        '内容': 30
    },
    'EMR-DISCHARGE': {
        '诊疗经过': 100,
        '出院医嘱': 20
    }
}

rules = min_length_rules[document.document_type]
FOR each field, min_length IN rules DO
    IF LENGTH(document[field]) < min_length THEN
        RETURN warning(field + "内容建议不少于" + min_length + "字符")
    END IF
END FOR

RETURN success()
``` |
| 失败处理 | 返回建议性提示,不阻止保存但影响质控评分 |

              WHERE admission_id = :admission_id 
                AND document_type = 'EMR-DAILY-PROGRESS'
              ORDER BY record_time DESC 
              LIMIT 1

IF last_record IS NOT NULL THEN
    IF new_record.record_time < last_record.record_time THEN
        RETURN error("病程记录时间不能早于上一条记录(" +

    END IF
END IF

// 检查是否在入院时间之后
admission = SELECT * FROM ip_admission WHERE admission_id = :admission_id

    RETURN error("病程记录时间不能早于入院时间")

#### BR-EMR-007: 诊断ICD-10编码校验

| 属性 | 内容 |
|------|------|
| 规则编号 | BR-EMR-007 |
| 规则名称 | 诊断ICD-10编码校验 |
| 规则类型 | VAL (校验规则) |
| 适用对象 | 诊断记录(diagnosis) |
| 规则描述 | 病历中的诊断必须关联有效的ICD-10编码 |
| 触发时机 | 录入诊断时 |
| 校验逻辑 | ```
// 伪代码
IF diagnose_code IS NOT NULL THEN
    icd10 = SELECT * FROM dict_icd10 WHERE code = :diagnose_code
    
    IF icd10 IS NULL THEN
        // 搜索相似诊断
        similar = SELECT * FROM dict_icd10 
                  WHERE name LIKE '%' + diagnose_keyword + '%' 
                  LIMIT 10
        RETURN error("未找到匹配的ICD-10编码", similar_diagnoses=similar)
    END IF
    
    // 检查编码有效性
    IF icd10.status != 'ACTIVE' THEN
        RETURN error("该诊断编码已停用,请选择有效编码")
    END IF
ELSE
    // 尝试自动匹配
    matched = SEARCH_icd10_by_name(diagnose_name)
    IF matched.count = 1 THEN
        // 自动关联唯一匹配的编码
        diagnose_code = matched[0].code
        RETURN success_with_auto_match(diagnose_code)
    ELSE IF matched.count > 1 THEN
        RETURN require_select("请选择正确的ICD-10编码", matched)
    ELSE
        RETURN error("未找到匹配的诊断编码,请手动选择")
    END IF
END IF

RETURN success()
``` |
| 失败处理 | 返回错误提示,提供相似诊断建议 |
| 优先级 | P0 |

| 适用对象 | 封存病历(emr_seal) |

| 触发时机 | 病历修改/删除操作时 |
| 校验逻辑 | ```
// 伪代码
// 检查病历封存状态
seal_record = SELECT * FROM emr_seal 
              WHERE archive_id = :archive_id 
                AND seal_status = 'SEALED'

IF seal_record IS NOT NULL THEN
    // 检查是否在封存期限内
    IF NOW() < seal_record.seal_time + seal_record.seal_duration_days DAYS THEN
        // 检查是否有解封审批
        unseal_approval = SELECT * FROM approval 
                          WHERE reference_id = seal_record.seal_id 
                            AND approval_type = 'UNSEAL'

        IF unseal_approval IS NULL THEN
            RETURN error("病历已封存,禁止操作。封存原因:" + seal_record.seal_reason + 
                        ";封存期限至:" + FORMAT_DATE(seal_record.seal_time + seal_record.seal_duration_days DAYS))
        END IF
    END IF
END IF

RETURN success()
``` |
| 失败处理 | 返回"病历已封存,禁止操作",阻止修改/删除 |

| 适用对象 | 病历借阅(emr_borrow) |

| 校验逻辑 | ```
// 伪代码
archive = SELECT * FROM emr_archive WHERE archive_id = :archive_id
borrower = SELECT * FROM user WHERE user_id = :borrower_id

// 检查是否为本科室病历
IF archive.dept_id != borrower.dept_id THEN
    // 非本科室病历需要额外审批
    borrow.need_extra_approval = true
    RETURN require_approval("借阅非本科室病历需要审批")
END IF

// 检查封存状态
IF archive.archive_status = 'SEALED' THEN
    IF NOT has_permission(borrower, 'BORROW_SEALED_EMR') THEN
        RETURN error("无权限借阅封存病历")
    END IF
    
    // 封存病历借阅需要更高级别审批
    borrow.approval_level = 'HIGH'
END IF

// 检查借阅类型权限
CASE borrow.borrow_type
    WHEN 'COPY' THEN  // 复印
        IF NOT has_permission(borrower, 'EMR_COPY') THEN
            RETURN error("无病历复印权限")
        END IF
    WHEN 'BORROW' THEN  // 借出
        IF NOT has_permission(borrower, 'EMR_BORROW') THEN
            RETURN error("无病历借出权限")
        END IF
END CASE

RETURN success()
``` |

    'EMR-ROUND-RECORD': {'frequency': 'WEEKLY', 'count_per_week': 2, 'deduction_per_miss': 5},
    'EMR-DISCHARGE': {'limit_hours': 48, 'deduction_per_hour': 5, 'max_deduction': 20},
    'EMR-DEATH': {'limit_hours': 24, 'deduction_per_hour': 10, 'max_deduction': 30}
}

admission = SELECT * FROM ip_admission WHERE admission_id = :admission_id

// 检查入院记录时限
admission_record = SELECT * FROM emr_document 
                   WHERE admission_id = :admission_id 
                     AND document_type = 'EMR-ADMISSION'
IF admission_record IS NOT NULL THEN
    hours_elapsed = HOUR(admission_record.record_time - admission.admission_time)
    IF hours_elapsed > time_limit_rules['EMR-ADMISSION'].limit_hours THEN
        overtime_hours = hours_elapsed - time_limit_rules['EMR-ADMISSION'].limit_hours
        deduction = MIN(overtime_hours * 5, time_limit_rules['EMR-ADMISSION'].max_deduction)
        deduction_score = deduction_score + deduction
        ADD qc_issue('时限', '入院记录超时' + overtime_hours + '小时', deduction)
    END IF
ELSE
    // 未完成入院记录
    hours_since_admission = HOUR(NOW() - admission.admission_time)
    IF hours_since_admission > 24 THEN
        deduction_score = deduction_score + 30
        ADD qc_issue('时限', '入院记录未完成', 30)
    END IF
END IF

// 检查首次病程记录时限
first_progress = SELECT * FROM emr_document

    hours_elapsed = HOUR(first_progress.record_time - admission.admission_time)
    IF hours_elapsed > 8 THEN
        overtime_hours = hours_elapsed - 8
        deduction = MIN(overtime_hours * 5, 20)
        deduction_score = deduction_score + deduction
        ADD qc_issue('时限', '首次病程记录超时' + overtime_hours + '小时', deduction)
    END IF
END IF

// 检查日常病程记录频率
stay_days = DATEDIFF(NOW(), admission.admission_date)
FOR day = 1 TO stay_days DO
    check_date = admission.admission_date + day DAYS
    daily_progress = SELECT * FROM emr_document 
                     WHERE admission_id = :admission_id 
                       AND document_type = 'EMR-DAILY-PROGRESS'
                       AND DATE(record_time) = check_date
    IF daily_progress IS NULL THEN
        deduction_score = deduction_score + 5
        ADD qc_issue('时限', FORMAT_DATE(check_date) + '缺少病程记录', 5)
    END IF
END FOR

// 检查出院记录时限(如果已出院)
IF admission.discharge_time IS NOT NULL THEN
    discharge_record = SELECT * FROM emr_document 
                       WHERE admission_id = :admission_id 
                         AND document_type = 'EMR-DISCHARGE'
    IF discharge_record IS NOT NULL THEN
        hours_elapsed = HOUR(discharge_record.record_time - admission.discharge_time)
        IF hours_elapsed > 48 THEN
            overtime_hours = hours_elapsed - 48
            deduction = MIN(overtime_hours * 5, 20)
            deduction_score = deduction_score + deduction
            ADD qc_issue('时限', '出院记录超时' + overtime_hours + '小时', deduction)
        END IF
    END IF
END IF

qc_score = 100 - deduction_score

RETURN {score: qc_score, deduction: deduction_score, issues: qc_issues}
